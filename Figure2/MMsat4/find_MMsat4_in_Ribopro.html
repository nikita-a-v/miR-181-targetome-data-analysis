OLgenes <- as.data.frame(subsetByOverlaps(gff23, MMSAT4))


# opposite strand
antiMMSAT4 <- MMSAT4
strand(antiMMSAT4) <- ifelse(strand(MMSAT4) == '+', '-', '+')

antiOLgenes <- as.data.frame(subsetByOverlaps(gff23, antiMMSAT4))
